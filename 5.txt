### 2. Client-Side Transit Map (JavaScript)
This script manages the dynamic elements of the authoring form, compiles the inputs into the expected fillable-form JSON structure, and transmits it to the backend ingestion API.
**Admin Workflow Script (admin_workflow.js)**
```javascript
/*
  NOTATION: Client-Side Data Compiler.
  USE: Orchestrates the admin form, handles UUID generation, packages the payload, and executes the POST request to the API.
*/

// Dynamically add relationship fields
function addRelationshipField() {
    const container = document.getElementById('relationships-container');
    const block = document.createElement('div');
    block.className = 'relationship-block';
    
    // Fillable form elements for edge mapping
    block.innerHTML = `
        <input type="text" class="rel-target" placeholder="Target UUID" style="width: 45%;" required>
        <select class="rel-type" style="width: 45%; margin-left: 5%;">
            <option value="implies">Implies</option>
            <option value="contradicts">Contradicts</option>
            <option value="extends">Extends</option>
            <option value="requires">Requires</option>
        </select>
        <button type="button" onclick="this.parentElement.remove()" style="margin-top: 5px; background: #f92672; color: white;">Remove</button>
    `;
    container.appendChild(block);
}

// Intercept form submission and build JSON map
document.getElementById('logic-author-form').addEventListener('submit', async function(e) {
    e.preventDefault();

    // Generate standard payload map
    const payload = {
        "_notation": "TEMPLATE: Fillable form for atomic logic artifact retrieval. Target for ingestion engines.",
        "artifact_id": document.getElementById('input-uuid').value || crypto.randomUUID(),
        "title": document.getElementById('input-title').value,
        "representations": {
            "condensed": document.getElementById('input-condensed').value,
            "verbose": document.getElementById('input-verbose').value,
            "foundational": {}
        },
        "interrelationships": []
    };

    // Safely parse foundational JSON if provided
    const rawFoundational = document.getElementById('input-foundational').value;
    if (rawFoundational.trim() !== '') {
        try {
            payload.representations.foundational = JSON.parse(rawFoundational);
        } catch (error) {
            alert("Foundational Data must be valid JSON.");
            return;
        }
    }

    // Extract relationships
    const relBlocks = document.querySelectorAll('.relationship-block');
    relBlocks.forEach(block => {
        payload.interrelationships.push({
            "target_id": block.querySelector('.rel-target').value,
            "type": block.querySelector('.rel-type').value,
            "uri": `/api/logic/${block.querySelector('.rel-target').value}`
        });
    });

    // Transmit to Backend Receiver
    try {
        const response = await fetch('/api/admin/ingest', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(payload)
        });
        
        if (response.ok) {
            alert('Artifact successfully committed to the database.');
            this.reset();
            document.getElementById('relationships-container').innerHTML = '';
        } else {
            alert('Ingestion pipeline rejected the payload.');
        }
    } catch (error) {
        console.error('Transit failure:', error);
    }
});

```

### 4. Authoring Workflow Map
To utilize these tools effectively, administrators or system users adhere to the following sequence to populate the graph:
 1. **Draft Node Identity:** The author opens the admin UI. They provide a clear, system-friendly Title and leave the Artifact UUID blank to allow the system to generate a unique cryptographic ID, unless they are updating an existing node.
 2. **Input Raw Logic:** The author writes the Verbose Description using complete sentences.
 3. **Cross-Reference Generation:** The author uses a secondary browser window mapped to the system's "Browse" frame to find UUIDs of existing logic that connects to this new concept.
 4. **Format Condensed Map:** The author writes the Condensed Logic. Whenever an existing concept is referenced, they use the exact notation format: [[UUID|Display Text]].
 5. **Append Edges:** The author clicks "+ Add Edge" for every required relationship, pasting the target UUIDs and selecting the relational vector (e.g., *implies*, *requires*).
 6. **Pipeline Commit:** Submitting the form compiles the artifacts into the JSON standard and fires it to the receiver. The node is immediately available in the UI without a system restart.
