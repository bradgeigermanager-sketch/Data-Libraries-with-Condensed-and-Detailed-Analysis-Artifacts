### Bidirectional Export Engine
To maintain the system as a decoupled ecosystem, data must flow out just as easily as it flows in. This engine reconstructs the atomic JSON artifacts from the relational graph, allowing for library backups, migrations, or ingestion by secondary systems.
```python
"""
NOTATION: Bidirectional JSON Export Engine.
USE: Reverses the ingestion process. Extracts a single logic node and its relationships from the SQL database and serializes it back into the standardized JSON machine-readable map.
"""

import json
import sqlite3

def export_logic_artifact(artifact_id: str, db_connection_string: str) -> str:
    """
    NOTATION: Serialization routine.
    USE: Reconstructs the fillable-form JSON format for a specific UUID.
    """
    conn = sqlite3.connect(db_connection_string)
    cursor = conn.cursor()
    
    try:
        # 1. Retrieve Core Artifact
        cursor.execute("""
            SELECT title, condensed_logic, verbose_desc, foundational_data 
            FROM LogicArtifact 
            WHERE artifact_id = ?
        """, (artifact_id,))
        core_row = cursor.fetchone()
        
        if not core_row:
            return "{}" # Return empty map if not found

        # 2. Retrieve Relationships
        cursor.execute("""
            SELECT target_id, relationship_type 
            FROM LogicRelationship 
            WHERE source_id = ?
        """, (artifact_id,))
        rel_rows = cursor.fetchall()
        
        # 3. Assemble JSON Map
        export_map = {
            "_notation": "TEMPLATE: Fillable form for atomic logic artifact retrieval. Target for ingestion engines.",
            "artifact_id": artifact_id,
            "title": core_row[0],
            "representations": {
                "condensed": core_row[1],
                "verbose": core_row[2],
                "foundational": json.loads(core_row[3]) if core_row[3] else {}
            },
            "interrelationships": [
                {
                    "target_id": rel[0],
                    "type": rel[1],
                    "uri": f"/api/logic/{rel[0]}"
                } for rel in rel_rows
            ]
        }
        
        return json.dumps(export_map, indent=2)
        
    finally:
        conn.close()

```
